    }
  }
  
  fn say_hi(&self) {
    println!("hello world!");
    println!("{}", self.pronouns);
    println!("My name is {}, and I am a {}", self.name, self.role);

    println!("I can code in the following programming languages:");
    for c in &self.code {
      println!("{}", c)
    }
  }
}

fn main() {
  let profile = Profile::new();
  profile.say_hi();
